window.onload = function() { const pedidoLocal = JSON.parse(localStorage.getItem("ultimoPedido")); const contenedor = document.getElementById("contenido-seguimiento"); if (!pedidoLocal || !pedidoLocal.id) { contenedor.innerHTML = `
No hay pedidos activos.
`; return; } // --- CORRECCIÓN DE HORA APROXIMADA --- // Si la propiedad 'hora' no existe, calculamos una basada en el momento actual + 45 min let horaEntrega = pedidoLocal.hora || pedidoLocal.entregaAprox; if (!horaEntrega) { const ahora = new Date(); ahora.setMinutes(ahora.getMinutes() + 45); horaEntrega = ahora.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }); } // ------------------------------------- let listaProdHTML = ""; if (pedidoLocal.productos) { // Convertimos a array por si viene como objeto desde Firebase const productosArray = Array.isArray(pedidoLocal.productos) ? pedidoLocal.productos : Object.values(pedidoLocal.productos); listaProdHTML = productosArray.map(p => `
${p.cantidad}x ${p.nombre || "Producto"} $${(p.precio || 0) * (p.cantidad || 0)}
`).join(""); } contenedor.innerHTML = `
Orden #${pedidoLocal.id} CONECTANDO...
ENTREGA APROX.
${horaEntrega}
TOTAL
$${pedidoLocal.total}
ENTREGAR EN:

${pedidoLocal.direccion || "No especificada"}

TUS PRODUCTOS:
${listaProdHTML}
`; const pedidoRef = ref(db, 'pedidos/' + pedidoLocal.id); onValue(pedidoRef, (snapshot) => { const data = snapshot.val(); if (data && data.estado) { if (data.estado.toLowerCase().trim() === "finalizado") { notificarAndroid("Tu pedido ha sido finalizado. ¡Gracias!"); contenedor.innerHTML = `

¡Pedido Finalizado!

Volver al Menú
`; localStorage.removeItem("ultimoPedido"); } else { actualizarUI(data.estado); } } }); };
